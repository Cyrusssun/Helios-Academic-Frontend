<script>
function switchTab(tab) {
  document.querySelectorAll('.lg-tab').forEach((t,i)=>t.classList.toggle('active',(i===0&&tab==='login')||(i===1&&tab==='signup')));
  document.getElementById('loginForm').classList.toggle('show', tab==='login');
  document.getElementById('signupForm').classList.toggle('show', tab==='signup');
}

function setFeedback(id, msg, type) {
  const el = document.getElementById(id);
  el.textContent = msg;
  el.className = 'lg-feedback ' + type;
}

function doLogin() {
  const email = document.getElementById('loginEmail').value.trim();
  const pass  = document.getElementById('loginPass').value.trim();
  if (!email || !pass) { setFeedback('loginFeedback','Please fill in all fields.','err'); return; }
  if (!email.includes('@')) { setFeedback('loginFeedback','Please enter a valid email.','err'); return; }
  // Simulate login
  setFeedback('loginFeedback','Signing you in...','ok');
  setTimeout(() => { window.location.href = '/practice/'; }, 800);
}

function doSignup() {
  const first = document.getElementById('signupFirst').value.trim();
  const last  = document.getElementById('signupLast').value.trim();
  const email = document.getElementById('signupEmail').value.trim();
  const pass  = document.getElementById('signupPass').value.trim();
  if (!first||!last||!email||!pass) { setFeedback('signupFeedback','Please fill in all fields.','err'); return; }
  if (pass.length < 8) { setFeedback('signupFeedback','Password must be at least 8 characters.','err'); return; }
  setFeedback('signupFeedback',`Welcome, ${first}! Redirecting...`,'ok');
  setTimeout(() => { window.location.href = '/practice/'; }, 1000);
}

  alert('Google OAuth would be configured here in production.');
}
